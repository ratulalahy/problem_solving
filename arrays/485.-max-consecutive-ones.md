```python
class Solution:
    def findMaxConsecutiveOnes(self, nums: List[int]) -> int:
        max_cons = 0
        cur_cons = 0
        
        for cur_item in nums:
            if cur_item == 1:
                cur_cons += 1
            else:
                max_cons = max(max_cons, cur_cons)
                cur_cons = 0
        return max(max_cons, cur_cons)
```
