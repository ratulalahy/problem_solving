class Solution:
    def findNumbers(self, nums: List[int]) -> int:
        counter_even = 0
        for cur_num in nums:
            counter_digit = 0
            while cur_num > 0:
                counter_digit += 1
                cur_num = cur_num//10
            counter_even += 1 if counter_digit % 2 == 0 else 0
        return counter_even
